Citing NCVX
============

    author  = {Buyun Liang and Ju Sun},
    title   = {NCVX: A User-Friendly and Scalable Package for Nonconvex Optimization in Machine Learning},
    journal = {arXiv preprint arXiv:XXXX.XXXXX},
    year    = {2021},
  }
